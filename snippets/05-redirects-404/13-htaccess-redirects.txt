# Permanent redirects via htaccess
#
# Faster than PHP because Apache processes htaccess before WordPress loads.
# Use this for large redirect maps (hundreds of entries) or when a redirect
# absolutely must run before PHP touches the request.
#
# Drop these BEFORE the WordPress block in .htaccess. Test on staging first.
#
# Workshop: WCEU 2026 — Do you really need an SEO/GEO plugin for WordPress?

<IfModule mod_rewrite.c>
    RewriteEngine On

    # 1) Single permanent redirects.
    Redirect 301 /old-page/ /new-page/
    Redirect 301 /services/seo/ /seo-services/
    Redirect 301 /blog/2024/welcome/ /welcome/

    # 2) Redirect a whole subdirectory.
    RedirectMatch 301 ^/blog/2023/(.*)$ /archive/$1

    # 3) Force HTTPS (most hosts already do this, useful as backup).
    RewriteCond %{HTTPS} !=on
    RewriteRule ^(.*)$ https://%{HTTP_HOST}/$1 [R=301,L]

    # 4) Force non-www to www (or the other way around — pick one).
    # RewriteCond %{HTTP_HOST} ^example\.com [NC]
    # RewriteRule ^(.*)$ https://www.example.com/$1 [R=301,L]

    # 5) Block trailing-slash inconsistencies.
    # RewriteCond %{REQUEST_FILENAME} !-f
    # RewriteCond %{REQUEST_URI} !(.*)/$
    # RewriteRule ^(.*)$ /$1/ [L,R=301]
</IfModule>
